\section{Day 37: Integration over Rectangles (Jan 07, 2026)}

\begin{definition}
    The lower and upper integral of $f$ on $R$ are defined by
    \[
        \underline{I_R}(f) = \sup_{P} L_P(f), \quad \overline{I_R}(f) = \inf_{P} L_P(f)
    \]
\end{definition}

\begin{simplethm}
    If $f : R \to \mathbb{R}$ is a bounded function, then both $\underline{I_R}(f)$ and $\overline{I_R}(f)$ exist, and $\underline{I_R}(f) \leq \overline{I_R}(f)$
\end{simplethm}

\begin{definition}[Darboux Integral]
    For bounded $f : R \to \mathbb{R}$,  if $\underline{I_R}(f) = \overline{I_R}(f)$ then $f$ is integrable on $R$ and the integral of $f$ on $R$ is defined by
    \[
    \int_R f \, dV := \underline{I_R}(f) = \overline{I_R}(f)
    \]
    otherwise $f$ is non-integrable.
\end{definition}

$dV$ denotes the volume element and is just notation.
