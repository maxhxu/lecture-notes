\section{Day 14: Global Extrema (Oct 06, 2025)}

\begin{definition}
    Have $f : A \subseteq \mathbb{R}^n \to \mathbb{R}$. Let $p \in A$.
    \begin{itemize}
        \item $p$ is a global maximum point of $f$ on $A$ if for all $x \in A$, $f(p) \geq f(x)$ 
        \item $p$ is a global minimum point of $f$ on $A$ if for all $x \in A$, $f(p) \leq f(x)$
    \end{itemize}
    We call $f(p)$ the global maximum/minimum value of $f$ on $A$, and say that $f$ attains a global maximum/minimum on $A$ respectively. 
\end{definition}

\begin{simplethm}[Single Variable EVT]
    Let $f : [a, b] \to \mathbb{R}$ be continuous. Then $f$ attains a maximum and a minimum on $[a, b]$. 
\end{simplethm}

\begin{simplethm}[EVT] \label{thm:evt}
    Have $A \subseteq \mathbb{R}^n$ be non-empty and compact, and $f : A \to \mathbb{R}$ be continuous. $f$ attains a maximum and minimum on $A$. 
\end{simplethm}
