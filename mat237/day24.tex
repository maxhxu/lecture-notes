\section{Day 24: Tangent Spaces (Nov 04, 2025)}

\begin{definition}[Tangent Vector]
Have $S \subseteq \mathbb{R}^n$, $p \in S$. For $v \in \mathbb{R}^n$, $v$ is a tangent vector of $S$ at $p$ if there exists an open interval $I \subseteq \mathbb{R}$, $0 \in I$, some $\gamma : I \to \mathbb{R}^n$, with $\gamma$ being $C^1$, $\gamma(I) \subseteq S$, $\gamma(0) = p$, and $\gamma'(0) = v$.
\end{definition}

\begin{definition}[Tangent Space]
The tangent space of $S$ at $p$, $T_PS$, is the set of all tangent vectors of $S$ at $p$.
\end{definition}

The tangent space always contains the zero vector, choose a $\gamma$ that always maps to $p$. The tangent plane may not.

\begin{definition}[Tangent Plane]
    Written $p + T_PS = \{ p + v : v \in T_PS \}$
\end{definition}
