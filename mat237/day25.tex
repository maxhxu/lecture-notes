\section{Day 25: Smooth Manifolds (Nov 06, 2025)}

\begin{definition}
Have $k, n \in \mathbb{N}^+$, $k < n$, with $S \subseteq \mathbb{R}^n$ and $p \in S$. $S$ is a $k$-dimensional smooth manifold at $p$, if there exists an open set $U \subseteq \mathbb{R}^n$, $p \in U$ such that
    $S \cap U$ is a graph of a $C^1$ function $f: V \to \mathbb{R}^{n-k}$, where $V$ is open. 
\end{definition}

A set $S \subseteq \mathbb{R}^n$ is a $k$-dimensional smooth manifold if $S$ is a $k$-dimensional smooth manifold at every point in $S$.

\begin{simplethm}
    Have $S, S' \subseteq \mathbb{R}^n$, with $p \in S \cap S'$. If there exists an open set $U \subseteq \mathbb{R}^n$ containing $p$ with $S \cap U = S' \cap U$, then $T_pS = TpS'$.
\end{simplethm}

For $v \in T_pS$, notice that there exists some smaller interval $I' \subseteq I$ with $p \in I'$ and $I' \subseteq S'$, since $p \in S \cap S'$ and both sets are open.

\begin{theorem}
    Let $k, n \in \mathbb{N}^+$ with $k < n$. Have $S \subseteq \mathbb{R}^n$, $p \in S$. If $S$ is a $k$-dimensional manifold at $p$, then $T_pS$ is a $k$-dimensional subspace of $\mathbb{R}^n$.
\end{theorem}
