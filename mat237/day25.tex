\section{Day 25: Smooth Manifolds (Nov 06, 2025)}

\begin{definition}
Have $k, n \in \mathbb{N}^+$, $k < n$, with $S \subseteq \mathbb{R}^n$ and $p \in S$. $S$ is a $k$-dimensional smooth manifold at $p$, if there exists an open set $U \subseteq \mathbb{R}^n$, $p \in U$ such that
    $S \cap U$ is a graph of a $C^1$ function $f: V \to \mathbb{R}^{n-k}$, where $V$ is open. 
\end{definition}
