\section{Day 36: Upper and Lower Sums (Jan 06, 2026)}
Let $R$ be a rectangle in $\mathbb{R}^n$, and let $f : R \to \mathbb{R}$ be a bounded function. Let $P$ be a partition of $R$ with index set $I$, with subrectangles $\{ R_i : i \in I \}$. Let $P'$ be a refinement of $P$.

\begin{definition}[Lower and Upper Sums]
   The $P$-lower and $P$-upper sums of $f$ are respectively given by
    \[
        L_P(f) = \sum_{i \in I} m_i \Vol(R_i), \quad U_P(f) = \sum_{i \in I} M_i \Vol(R_i)
    \]
    where for $i \in I$,
    \[
        m_i = \inf \{ f(x) : x \in R_i \}, \quad M_i = \sup \{ f(x) : x \in R_i \}
    \]
\end{definition}

\begin{definition}[Riemann Sum]
    For each $i \in I$, a sample point $x_i^* \in R_i$ is chosen. The Riemann sum for $f$ with $P$ and these sample points is given by
    \[
        S_P^*(f) = \sum_{i \in I} f(x_i^*)\Vol(R_i)
    \]
\end{definition}

\begin{theorem}
    Properties of partitions include:
    \begin{itemize}
        \item $L_P(f) \leq U_P(f)$
        \item $L_P(f) \leq L_{P'}(f)$, $U_{P'}(f) \leq U_P(f)$
        \item For any two partitions $A, B$ of a rectangle, $L_A(f) \leq U_B(f)$
        \item $U_P(f + g) \leq U_P(f) + U_P(g)$
        \item For $\lambda > 0$, $U_P(\lambda f) = \lambda U_P(f)$
        \item $U_P(-f) = - L_P(f)$
        \item If $f \leq g$ on $R$, then $U_P(f) \leq U_P(g)$
        \item $S_P^*(f + \lambda g) = S_P^*(f) + \lambda S_P^*(g)$
        \item If $f \leq g$ on $R$, then $S_P^*(f) \leq S_P^*(g)$
    \end{itemize}
\end{theorem}
